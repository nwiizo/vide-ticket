3use crate::Result;

8/// MCP event handler that processes CLI events
9pub struct McpEventHandler {}
10
11impl McpEventHandler {
12    /// Create a new MCP event handler
13    pub fn new(_service: Arc<VibeTicketService>) -> Self {
14        Self {}
15    }
16}

18#[async_trait::async_trait]
19impl EventHandler for McpEventHandler {
20    async fn handle_event(&self, event: TicketEvent) -> Result<()> {
21        match event {
22            TicketEvent::Created(ticket) => {
23                // Clear any cached ticket lists
24                tracing::info!("MCP: Ticket created via CLI: {}", ticket.id);
25                // In the future, we could notify connected MCP clients here
26            },
27            TicketEvent::Updated(ticket) => {
28                tracing::info!("MCP: Ticket updated via CLI: {}", ticket.id);
29                // Update any internal caches if needed
30            },
31            TicketEvent::Closed(ticket_id, message) => {
32                tracing::info!("MCP: Ticket closed via CLI: {} - {}", ticket_id, message);
33            },
34            TicketEvent::TaskAdded(ticket_id, task) => {
35                tracing::info!(
36                    "MCP: Task added via CLI to ticket {}: {}",
37                    ticket_id,
38                    task.title
39                );
40            },
41            TicketEvent::TaskCompleted(ticket_id, task_id) => {
42                tracing::info!(
43                    "MCP: Task completed via CLI: {} in ticket {}",
44                    task_id,
45                    ticket_id
46                );
47            },
48            TicketEvent::TaskRemoved(ticket_id, task_id) => {
49                tracing::info!(
50                    "MCP: Task removed via CLI: {} from ticket {}",
51                    task_id,
52                    ticket_id
53                );
54            },
55            TicketEvent::StatusChanged(ticket_id, old_status, new_status) => {
56                tracing::info!(
57                    "MCP: Ticket {} status changed via CLI: {:?} -> {:?}",
58                    ticket_id,
59                    old_status,
60                    new_status
61                );
62            },
63        }
64
65        Ok(())
66    }
67}
