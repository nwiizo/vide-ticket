4/// Represents the current status of a ticket
5///
6/// The status follows a typical workflow progression from
7/// Todo â†’ Doing â†’ Done, with additional states for special cases.
8#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq, Hash)]
9#[serde(rename_all = "lowercase")]
10pub enum Status {
11    /// Ticket is created but work hasn't started
12    Todo,
13
14    /// Work is actively being done on the ticket
15    Doing,
16
17    /// Work on the ticket is completed
18    Done,
19
20    /// Ticket is blocked by external dependencies
21    Blocked,
22
23    /// Ticket is in review/QA phase
24    Review,
25}
26

77    /// Returns whether the status represents active work

79        matches!(self, Self::Doing | Self::Review)

82    /// Returns whether the status represents completed work

84        matches!(self, Self::Done)

115impl TryFrom<&str> for Status {
116    type Error = String;
117
118    fn try_from(value: &str) -> Result<Self, Self::Error> {
119        match value.to_lowercase().as_str() {
120            "todo" => Ok(Self::Todo),
121            "doing" | "in-progress" | "wip" => Ok(Self::Doing),
122            "done" | "completed" | "closed" => Ok(Self::Done),
123            "blocked" => Ok(Self::Blocked),
124            "review" | "reviewing" => Ok(Self::Review),
125            _ => Err(format!("Invalid status: {value}")),
126        }
127    }
128}
