7/// Represents a ticket in the vibe-ticket system
8///
9/// A ticket encapsulates a unit of work with associated metadata,
10/// tasks, and lifecycle information.
11#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
12pub struct Ticket {
13    /// Unique identifier for the ticket
14    pub id: TicketId,
15
16    /// URL-friendly slug derived from the title
17    pub slug: String,
18
19    /// Human-readable title of the ticket
20    pub title: String,
21
22    /// Detailed description of the work to be done
23    pub description: String,
24
25    /// Priority level of the ticket
26    pub priority: Priority,
27
28    /// Current status of the ticket
29    pub status: Status,
30
31    /// Tags for categorization and filtering
32    #[serde(default)]
33    pub tags: Vec<String>,
34
35    /// Timestamp when the ticket was created
36    pub created_at: DateTime<Utc>,
37
38    /// Timestamp when work started on the ticket
39    pub started_at: Option<DateTime<Utc>>,
40
41    /// Timestamp when the ticket was closed
42    pub closed_at: Option<DateTime<Utc>>,
43
44    /// Username of the person assigned to the ticket
45    pub assignee: Option<String>,
46
47    /// List of tasks associated with this ticket
48    #[serde(default)]
49    pub tasks: Vec<Task>,
50
51    /// Additional metadata for extensibility
52    #[serde(default)]
53    pub metadata: HashMap<String, serde_json::Value>,
54}
55
56impl Ticket {
57    /// Creates a new ticket with the given slug and title
58    pub fn new(slug: impl Into<String>, title: impl Into<String>) -> Self {
59        Self {
60            id: TicketId::new(),
61            slug: slug.into(),
62            title: title.into(),
63            description: String::new(),
64            priority: Priority::default(),
65            status: Status::default(),
66            tags: Vec::new(),
67            created_at: Utc::now(),
68            started_at: None,
69            closed_at: None,
70            assignee: None,
71            tasks: Vec::new(),
72            metadata: HashMap::new(),
73        }
74    }
75
76    /// Creates a new ticket with a specific ID (useful for deserialization)
77    pub fn with_id(id: TicketId, slug: impl Into<String>, title: impl Into<String>) -> Self {
78        Self {
79            id,
80            slug: slug.into(),
81            title: title.into(),
82            description: String::new(),
83            priority: Priority::default(),
84            status: Status::default(),
85            tags: Vec::new(),
86            created_at: Utc::now(),
87            started_at: None,
88            closed_at: None,
89            assignee: None,
90            tasks: Vec::new(),
91            metadata: HashMap::new(),
92        }
93    }
94
95    /// Starts work on the ticket, updating status and timestamp
96    pub fn start(&mut self) {
97        self.status = Status::Doing;
98        self.started_at = Some(Utc::now());
99    }
100
101    /// Closes the ticket, updating status and timestamp
102    pub fn close(&mut self) {
103        self.status = Status::Done;
104        self.closed_at = Some(Utc::now());
105    }
106
107    /// Adds a task to the ticket
108    pub fn add_task(&mut self, title: impl Into<String>) -> TaskId {
109        let task = Task::new(title);
110        let task_id = task.id.clone();
111        self.tasks.push(task);
112        task_id
113    }
114
115    /// Marks a task as completed
116    pub fn complete_task(&mut self, task_id: &TaskId) -> Result<(), String> {
117        self.tasks
118            .iter_mut()
119            .find(|task| &task.id == task_id)
120            .ok_or_else(|| format!("Task with ID {task_id} not found"))?
121            .complete();
122        Ok(())
123    }
124
125    /// Returns the number of completed tasks
126    pub fn completed_tasks_count(&self) -> usize {
127        self.tasks.iter().filter(|task| task.completed).count()
128    }
129
130    /// Returns the total number of tasks
131    pub fn total_tasks_count(&self) -> usize {
132        self.tasks.len()
133    }
134
135    /// Calculates the completion percentage
136    pub fn completion_percentage(&self) -> f32 {
137        if self.tasks.is_empty() {
138            0.0
139        } else {
140            (self.completed_tasks_count() as f32 / self.total_tasks_count() as f32) * 100.0
141        }
142    }
143
144    /// Returns the duration the ticket has been open
145    pub fn duration(&self) -> chrono::Duration {
146        let end_time = self.closed_at.unwrap_or_else(Utc::now);
147        end_time - self.created_at
148    }
149
150    /// Returns the working duration (from start to close/now)
151    pub fn working_duration(&self) -> Option<chrono::Duration> {
152        self.started_at.map(|start| {
153            let end_time = self.closed_at.unwrap_or_else(Utc::now);
154            end_time - start
155        })
156    }
157}
