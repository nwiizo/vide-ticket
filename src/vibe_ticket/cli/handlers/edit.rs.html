11/// Handler for the `edit` command
12///
13/// This function allows editing various properties of a ticket:
14/// 1. Title
15/// 2. Description
16/// 3. Priority
17/// 4. Status
18/// 5. Tags (add/remove)
19/// 6. Opens in editor if requested
20///
21/// # Arguments
22///
23/// * `ticket_ref` - Optional ticket ID or slug (defaults to active ticket)
24/// * `title` - New title for the ticket
25/// * `description` - New description for the ticket
26/// * `priority` - New priority for the ticket
27/// * `status` - New status for the ticket
28/// * `add_tags` - Tags to add (comma-separated)
29/// * `remove_tags` - Tags to remove (comma-separated)
30/// * `editor` - Whether to open in the default editor
31/// * `project_dir` - Optional project directory path
32/// * `output` - Output formatter for displaying results
33///
34/// # Errors
35///
36/// Returns an error if:
37/// - The project is not initialized
38/// - No ticket is specified and there's no active ticket
39/// - The ticket is not found
40/// - Invalid priority or status values are provided
41#[allow(clippy::too_many_arguments)]
42pub fn handle_edit_command(
43    ticket_ref: Option<String>,
44    title: Option<String>,
45    description: Option<String>,
46    priority: Option<String>,
47    status: Option<String>,
48    add_tags: Option<String>,
49    remove_tags: Option<String>,
50    editor: bool,
51    project_dir: Option<&str>,
52    output: &OutputFormatter,
53) -> Result<()> {
54    // Ensure project is initialized
55    let project_root = find_project_root(project_dir)?;
56    let vibe_ticket_dir = project_root.join(".vibe-ticket");
57
58    // Initialize storage
59    let storage = FileStorage::new(&vibe_ticket_dir);
60
61    // Get the active ticket if no ticket specified
62    let ticket_id = if let Some(ref_str) = ticket_ref {
63        resolve_ticket_ref(&storage, &ref_str)?
64    } else {
65        // Get active ticket
66        storage
67            .get_active()?
68            .ok_or(VibeTicketError::NoActiveTicket)?
69    };
70
71    // Load the ticket
72    let mut ticket = storage.load(&ticket_id)?;
73
74    // Track what was changed
75    let mut changes = Vec::new();
76
77    // Open in editor if requested
78    if editor {
79        edit_in_editor(&mut ticket, &storage, output)?;
80        return Ok(());
81    }
82
83    // Update title if provided
84    if let Some(new_title) = title {
85        let old_title = ticket.title.clone();
86        ticket.title.clone_from(&new_title);
87        changes.push(format!("Title: {old_title} → {new_title}"));
88    }
89
90    // Update description if provided
91    if let Some(new_description) = description {
92        ticket.description = new_description;
93        changes.push("Description updated".to_string());
94    }
95
96    // Update priority if provided
97    if let Some(priority_str) = priority {
98        let new_priority = Priority::try_from(priority_str.as_str()).map_err(|_| {
99            VibeTicketError::InvalidPriority {
100                priority: priority_str,
101            }
102        })?;
103        let old_priority = ticket.priority;
104        ticket.priority = new_priority;
105        changes.push(format!("Priority: {old_priority} → {new_priority}"));
106    }
107
108    // Update status if provided
109    if let Some(status_str) = status {
110        let new_status = Status::try_from(status_str.as_str())
111            .map_err(|_| VibeTicketError::InvalidStatus { status: status_str })?;
112        let old_status = ticket.status;
113        ticket.status = new_status;
114        changes.push(format!("Status: {old_status} → {new_status}"));
115
116        // Update timestamps based on status changes
117        match (old_status, new_status) {
118            (Status::Todo, Status::Doing) => {
119                ticket.started_at = Some(chrono::Utc::now());
120            },
121            (_, Status::Done) if old_status != Status::Done => {
122                ticket.closed_at = Some(chrono::Utc::now());
123            },
124            _ => {},
125        }
126    }
127
128    // Add tags if provided
129    if let Some(tags_str) = add_tags {
130        let new_tags: Vec<String> = tags_str
131            .split(',')
132            .map(|s| s.trim().to_string())
133            .filter(|s| !s.is_empty())
134            .collect();
135
136        for tag in new_tags {
137            if !ticket.tags.contains(&tag) {
138                ticket.tags.push(tag);
139            }
140        }
141        changes.push("Tags added".to_string());
142    }
143
144    // Remove tags if provided
145    if let Some(tags_str) = remove_tags {
146        let tags_to_remove: Vec<String> = tags_str
147            .split(',')
148            .map(|s| s.trim().to_string())
149            .filter(|s| !s.is_empty())
150            .collect();
151
152        ticket.tags.retain(|tag| !tags_to_remove.contains(tag));
153        changes.push("Tags removed".to_string());
154    }
155
156    // Check if any changes were made
157    if changes.is_empty() {
158        output.warning("No changes specified");
159        return Ok(());
160    }
161
162    // Save the updated ticket
163    storage.save(&ticket)?;
164
165    // Output results
166    if output.is_json() {
167        output.print_json(&serde_json::json!({
168            "status": "success",
169            "ticket": {
170                "id": ticket.id.to_string(),
171                "slug": ticket.slug,
172                "title": ticket.title,
173                "description": ticket.description,
174                "status": ticket.status.to_string(),
175                "priority": ticket.priority.to_string(),
176                "tags": ticket.tags,
177            },
178            "changes": changes,
179        }))?;
180    } else {
181        output.success(&format!("Updated ticket: {}", ticket.slug));
182        for change in &changes {
183            output.info(&format!("  • {change}"));
184        }
185
186        // Show current state
187        output.info("");
188        output.info("Current state:");
189        output.info(&format!("  Title: {}", ticket.title));
190        output.info(&format!("  Status: {}", ticket.status));
191        output.info(&format!("  Priority: {}", ticket.priority));
192        if !ticket.tags.is_empty() {
193            output.info(&format!("  Tags: {}", ticket.tags.join(", ")));
194        }
195    }
196
197    Ok(())
198}

200/// Edit ticket in the default editor
201fn edit_in_editor(
202    ticket: &mut crate::core::Ticket,
203    storage: &FileStorage,
204    output: &OutputFormatter,
205) -> Result<()> {
206    use std::io::Write as IoWrite;
207    use std::process::Command;
208
209    // Create a temporary file with the ticket content
210    let temp_dir = std::env::temp_dir();
211    let temp_file = temp_dir.join(format!("vibe-ticket-{}.yaml", ticket.id));
212
213    // Serialize ticket to YAML
214    let yaml_content = serde_yaml::to_string(&ticket)
215        .map_err(|e| VibeTicketError::custom(format!("Failed to serialize YAML: {}", e)))?;
216
217    // Write to temporary file
218    let mut file = std::fs::File::create(&temp_file)
219        .map_err(|e| VibeTicketError::io_error("create", &temp_file, e))?;
220    file.write_all(yaml_content.as_bytes())
221        .map_err(|e| VibeTicketError::io_error("write", &temp_file, e))?;
222
223    // Get editor from environment
224    let editor = std::env::var("EDITOR").unwrap_or_else(|_| "vi".to_string());
225
226    // Open editor
227    let status = Command::new(&editor)
228        .arg(&temp_file)
229        .status()
230        .map_err(|e| VibeTicketError::custom(format!("Failed to launch editor: {e}")))?;
231
232    if !status.success() {
233        return Err(VibeTicketError::custom("Editor exited with error"));
234    }
235
236    // Read the edited content
237    let edited_content = std::fs::read_to_string(&temp_file)
238        .map_err(|e| VibeTicketError::io_error("read", &temp_file, e))?;
239
240    // Parse the edited ticket
241    let edited_ticket: crate::core::Ticket = serde_yaml::from_str(&edited_content)
242        .map_err(|e| VibeTicketError::deserialization_error("YAML ticket", e))?;
243
244    // Update the original ticket
245    *ticket = edited_ticket;
246
247    // Save the updated ticket
248    storage.save(ticket)?;
249
250    // Clean up temp file
251    let _ = std::fs::remove_file(&temp_file);
252
253    output.success(&format!("Updated ticket: {}", ticket.slug));
254
255    Ok(())
256}
