23/// Common metadata structure for JSON and YAML exports
24#[derive(Debug, Serialize)]
25pub struct ExportMetadata {
26    pub tickets: Vec<Ticket>,
27    pub exported_at: DateTime<Utc>,
28    pub total: usize,
29}
30
31impl ExportMetadata {
32    /// Create new export metadata with the given tickets
33    pub fn new(tickets: Vec<Ticket>) -> Self {
34        let total = tickets.len();
35        Self {
36            tickets,
37            exported_at: Utc::now(),
38            total,
39        }
40    }
41}

43/// Trait for ticket exporters
44pub trait Exporter {
45    /// Export tickets to the target format
46    fn export(&self, tickets: &[Ticket]) -> Result<String>;
47
48    /// Get the format name for display
49    fn format_name(&self) -> &'static str;
50}
51
52/// Handler for the `export` command
53///
54/// Exports tickets to various formats using the appropriate exporter
55pub fn handle_export_command(
56    format: &str,
57    output_path: Option<String>,
58    include_archived: bool,
59    project_dir: Option<&str>,
60    output: &OutputFormatter,
61) -> Result<()> {
62    // Get exporter for the format
63    let exporter: Box<dyn Exporter> = match format.to_lowercase().as_str() {
64        "json" => Box::new(JsonExporter),
65        "yaml" => Box::new(YamlExporter),
66        "csv" => Box::new(CsvExporter),
67        "markdown" | "md" => Box::new(MarkdownExporter),
68        _ => {
69            return Err(VibeTicketError::custom(format!(
70                "Unsupported export format: {format}. Supported formats: json, yaml, csv, markdown"
71            )));
72        },
73    };
74
75    // Load and filter tickets
76    let tickets = load_tickets(project_dir, include_archived)?;
77
78    // Export using the appropriate exporter
79    let content = exporter.export(&tickets)?;
80
81    // Output results
82    output_results(
83        content,
84        output_path,
85        tickets.len(),
86        exporter.format_name(),
87        include_archived,
88        output,
89    )
90}
91
92/// Load tickets from storage
93fn load_tickets(project_dir: Option<&str>, include_archived: bool) -> Result<Vec<Ticket>> {
94    let project_root = find_project_root(project_dir)?;
95    let vibe_ticket_dir = project_root.join(".vibe-ticket");
96    let storage = FileStorage::new(&vibe_ticket_dir);
97
98    let mut tickets = storage.load_all()?;
99
100    // Filter out archived tickets if not included
101    if !include_archived {
102        tickets.retain(|t| {
103            !t.metadata
104                .get("archived")
105                .and_then(serde_json::Value::as_bool)
106                .unwrap_or(false)
107        });
108    }
109
110    // Sort tickets by creation date
111    tickets.sort_by(|a, b| a.created_at.cmp(&b.created_at));
112
113    Ok(tickets)
114}
115
116/// Output export results
117fn output_results(
118    content: String,
119    output_path: Option<String>,
120    ticket_count: usize,
121    format_name: &str,
122    include_archived: bool,
123    output: &OutputFormatter,
124) -> Result<()> {
125    if let Some(path) = output_path {
126        std::fs::write(&path, content)
127            .map_err(|e| VibeTicketError::io_error("write", std::path::Path::new(&path), e))?;
128
129        output.success(&format!("Exported {ticket_count} tickets to {path}"));
130        output.info(&format!("Format: {format_name}"));
131        if !include_archived {
132            output.info(
133                "Note: Archived tickets were excluded. Use --include-archived to include them.",
134            );
135        }
136    } else {
137        // Output to stdout
138        println!("{content}");
139    }
140
141    Ok(())
142}
