9/// Markdown exporter implementation
10pub struct MarkdownExporter;
11
12impl Exporter for MarkdownExporter {
13    fn export(&self, tickets: &[Ticket]) -> Result<String> {
14        let mut output = String::new();
15
16        // Write header
17        write_header(&mut output, tickets.len());
18
19        // Write summary
20        write_summary(&mut output, tickets);
21
22        // Write tickets grouped by status
23        write_tickets_by_status(&mut output, tickets);
24
25        Ok(output)
26    }
27
28    fn format_name(&self) -> &'static str {
29        "Markdown"
30    }
31}

33/// Write the document header
34fn write_header(output: &mut String, ticket_count: usize) {
35    writeln!(output, "# Ticket Export\n").unwrap();
36    writeln!(
37        output,
38        "**Exported at**: {}",
39        chrono::Local::now().format("%Y-%m-%d %H:%M")
40    )
41    .unwrap();
42    writeln!(output, "**Total tickets**: {ticket_count}\n").unwrap();
43}
44
45/// Write the summary table
46fn write_summary(output: &mut String, tickets: &[Ticket]) {
47    writeln!(output, "## Summary\n").unwrap();
48    writeln!(output, "| Status | Count |").unwrap();
49    writeln!(output, "|--------|-------|").unwrap();
50
51    let status_counts = count_by_status(tickets);
52    for (status, count) in &status_counts {
53        writeln!(output, "| {status} | {count} |").unwrap();
54    }
55
56    writeln!(output, "\n## Tickets\n").unwrap();
57}
58
59/// Count tickets by status
60fn count_by_status(tickets: &[Ticket]) -> HashMap<String, usize> {
61    let mut counts = HashMap::new();
62    for ticket in tickets {
63        *counts.entry(ticket.status.to_string()).or_insert(0) += 1;
64    }
65    counts
66}
67
68/// Write tickets grouped by status
69fn write_tickets_by_status(output: &mut String, tickets: &[Ticket]) {
70    let groups = group_by_status(tickets);
71
72    if let Some(tickets) = groups.get(&Status::Todo) {
73        write_status_section(output, "ðŸ“‹ Todo", tickets);
74    }
75
76    if let Some(tickets) = groups.get(&Status::Doing) {
77        write_status_section(output, "ðŸ”„ In Progress", tickets);
78    }
79
80    if let Some(tickets) = groups.get(&Status::Review) {
81        write_status_section(output, "ðŸ‘€ In Review", tickets);
82    }
83
84    if let Some(tickets) = groups.get(&Status::Blocked) {
85        write_status_section(output, "ðŸš« Blocked", tickets);
86    }
87
88    if let Some(tickets) = groups.get(&Status::Done) {
89        write_status_section(output, "âœ… Done", tickets);
90    }
91}
92
93/// Group tickets by status
94fn group_by_status(tickets: &[Ticket]) -> HashMap<Status, Vec<&Ticket>> {
95    let mut groups: HashMap<Status, Vec<&Ticket>> = HashMap::new();
96
97    for ticket in tickets {
98        groups.entry(ticket.status).or_default().push(ticket);
99    }
100
101    groups
102}
103
104/// Write a section for a specific status
105fn write_status_section(output: &mut String, title: &str, tickets: &[&Ticket]) {
106    writeln!(output, "### {title}\n").unwrap();
107    for ticket in tickets {
108        write_ticket(output, ticket);
109    }
110}
111
112/// Write a single ticket in Markdown format
113fn write_ticket(output: &mut String, ticket: &Ticket) {
114    writeln!(output, "#### {} - {}\n", ticket.slug, ticket.title).unwrap();
115    writeln!(output, "- **Priority**: {}", ticket.priority).unwrap();
116
117    if let Some(assignee) = &ticket.assignee {
118        writeln!(output, "- **Assignee**: {assignee}").unwrap();
119    }
120
121    if !ticket.tags.is_empty() {
122        writeln!(output, "- **Tags**: {}", ticket.tags.join(", ")).unwrap();
123    }
124
125    if !ticket.tasks.is_empty() {
126        let completed = ticket.tasks.iter().filter(|t| t.completed).count();
127        writeln!(output, "- **Tasks**: {}/{}", completed, ticket.tasks.len()).unwrap();
128    }
129
130    writeln!(
131        output,
132        "- **Created**: {}",
133        ticket.created_at.format("%Y-%m-%d")
134    )
135    .unwrap();
136
137    if !ticket.description.trim().is_empty() {
138        writeln!(output).unwrap();
139        writeln!(output, "{}", ticket.description).unwrap();
140    }
141
142    writeln!(output, "\n---\n").unwrap();
143}
