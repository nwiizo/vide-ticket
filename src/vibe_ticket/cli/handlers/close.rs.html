12/// Handler for the `close` command
13///
14/// This function performs the following operations:
15/// 1. Loads the specified ticket (or active ticket if none specified)
16/// 2. Updates the ticket status to "done"
17/// 3. Sets the `closed_at` timestamp
18/// 4. Clears the active ticket if it was the one being closed
19/// 5. Optionally archives the ticket
20/// 6. Optionally creates a pull request
21///
22/// # Arguments
23///
24/// * `ticket_ref` - Optional ticket ID or slug (defaults to active ticket)
25/// * `message` - Optional close message
26/// * `archive` - Whether to archive the ticket
27/// * `create_pr` - Whether to create a pull request
28/// * `project_dir` - Optional project directory path
29/// * `output` - Output formatter for displaying results
30///
31/// # Errors
32///
33/// Returns an error if:
34/// - The project is not initialized
35/// - No ticket is specified and there's no active ticket
36/// - The ticket is not found
37/// - The ticket is already closed
38pub fn handle_close_command(
39    ticket_ref: Option<String>,
40    message: Option<String>,
41    archive: bool,
42    create_pr: bool,
43    project_dir: Option<&str>,
44    output: &OutputFormatter,
45) -> Result<()> {
46    // Ensure project is initialized
47    let project_root = find_project_root(project_dir)?;
48    let vibe_ticket_dir = project_root.join(".vibe-ticket");
49
50    // Initialize storage
51    let storage = FileStorage::new(&vibe_ticket_dir);
52
53    // Get the active ticket if no ticket specified
54    let ticket_id = if let Some(ref_str) = ticket_ref {
55        resolve_ticket_ref(&storage, &ref_str)?
56    } else {
57        // Get active ticket
58        storage
59            .get_active()?
60            .ok_or(VibeTicketError::NoActiveTicket)?
61    };
62
63    // Load the ticket
64    let mut ticket = storage.load(&ticket_id)?;
65
66    // Check if ticket is already closed
67    if ticket.status == Status::Done {
68        return Err(VibeTicketError::custom(format!(
69            "Ticket '{}' is already closed",
70            ticket.slug
71        )));
72    }
73
74    // Update ticket status and close time
75    let previous_status = ticket.status;
76    ticket.status = Status::Done;
77    ticket.closed_at = Some(Utc::now());
78
79    // Add close message to metadata if provided
80    if let Some(msg) = &message {
81        ticket.metadata.insert(
82            "close_message".to_string(),
83            serde_json::Value::String(msg.clone()),
84        );
85    }
86
87    // Save the updated ticket
88    storage.save(&ticket)?;
89
90    // Clear active ticket if this was the active one
91    if let Some(active_id) = storage.get_active()? {
92        if active_id == ticket_id {
93            storage.clear_active()?;
94        }
95    }
96
97    // Create pull request if requested
98    if create_pr {
99        create_pull_request(&project_root, &ticket, output)?;
100    }
101
102    // Archive if requested (for now, just add a flag to metadata)
103    if archive {
104        // In a real implementation, we might move the ticket to an archive directory
105        let mut archived_ticket = ticket.clone();
106        archived_ticket
107            .metadata
108            .insert("archived".to_string(), serde_json::Value::Bool(true));
109        archived_ticket.metadata.insert(
110            "archived_at".to_string(),
111            serde_json::Value::String(Utc::now().to_rfc3339()),
112        );
113        storage.save(&archived_ticket)?;
114    }
115
116    // Output results
117    if output.is_json() {
118        output.print_json(&serde_json::json!({
119            "status": "success",
120            "ticket": {
121                "id": ticket.id.to_string(),
122                "slug": ticket.slug,
123                "title": ticket.title,
124                "status": ticket.status.to_string(),
125                "closed_at": ticket.closed_at,
126                "archived": archive,
127            },
128            "message": message,
129            "pr_created": create_pr,
130        }))?;
131    } else {
132        output.success(&format!("Closed ticket: {}", ticket.slug));
133        output.info(&format!("Title: {}", ticket.title));
134        output.info(&format!("Status: {} → {}", previous_status, Status::Done));
135
136        if let Some(msg) = message {
137            output.info(&format!("Close message: {msg}"));
138        }
139
140        if archive {
141            output.info("Ticket has been archived");
142        }
143
144        if create_pr {
145            output.info("Pull request creation initiated");
146        }
147
148        // Calculate duration if started_at is available
149        if let Some(started_at) = ticket.started_at {
150            if let Some(closed_at) = ticket.closed_at {
151                let duration = closed_at - started_at;
152                let hours = duration.num_hours();
153                let minutes = duration.num_minutes() % 60;
154                output.info(&format!("\nTime spent: {hours}h {minutes}m"));
155            }
156        }
157    }
158
159    Ok(())
160}

162/// Create a pull request for the ticket
163fn create_pull_request(
164    project_root: &std::path::Path,
165    ticket: &crate::core::Ticket,
166    output: &OutputFormatter,
167) -> Result<()> {
168    use std::process::Command;
169
170    // Get current branch name
171    let current_branch = Command::new("git")
172        .arg("rev-parse")
173        .arg("--abbrev-ref")
174        .arg("HEAD")
175        .current_dir(project_root)
176        .output()
177        .map_err(|e| VibeTicketError::custom(format!("Failed to get current branch: {e}")))?;
178
179    if !current_branch.status.success() {
180        return Err(VibeTicketError::custom("Failed to get current branch name"));
181    }
182
183    let branch_name = String::from_utf8_lossy(&current_branch.stdout)
184        .trim()
185        .to_string();
186
187    // Check if we have the GitHub CLI installed
188    let gh_check = Command::new("gh").arg("--version").output();
189
190    if gh_check.is_err() || !gh_check.unwrap().status.success() {
191        output.warning(
192            "GitHub CLI (gh) not found. Please install it to create pull requests automatically.",
193        );
194        output.info(&format!(
195            "You can create a pull request manually for branch: {branch_name}"
196        ));
197        return Ok(());
198    }
199
200    // Create PR using GitHub CLI
201    let pr_title = format!("[{}] {}", ticket.slug, ticket.title);
202    let pr_body = format!(
203        "## Ticket: {}\n\n{}\n\n**Status:** {} → Done\n**Priority:** {}\n",
204        ticket.slug,
205        ticket.description,
206        if ticket.started_at.is_some() {
207            "Doing"
208        } else {
209            "Todo"
210        },
211        ticket.priority
212    );
213
214    let create_pr = Command::new("gh")
215        .arg("pr")
216        .arg("create")
217        .arg("--title")
218        .arg(&pr_title)
219        .arg("--body")
220        .arg(&pr_body)
221        .arg("--head")
222        .arg(&branch_name)
223        .current_dir(project_root)
224        .output()
225        .map_err(|e| VibeTicketError::custom(format!("Failed to create PR: {e}")))?;
226
227    if create_pr.status.success() {
228        let pr_url = String::from_utf8_lossy(&create_pr.stdout)
229            .trim()
230            .to_string();
231        output.success(&format!("Created pull request: {pr_url}"));
232    } else {
233        let error_msg = String::from_utf8_lossy(&create_pr.stderr);
234        output.warning(&format!("Failed to create pull request: {error_msg}"));
235        output.info("You can create the pull request manually");
236    }
237
238    Ok(())
239}
